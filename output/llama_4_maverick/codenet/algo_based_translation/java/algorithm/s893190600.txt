public void solve() {
    int H = ni();
    int W = ni();
    char[][] C = new char[H + 2][W + 2];
    char[] c;
    for (int i = 0; i < H; i++) {
        c = ns().toCharArray();
        for (int j = 0; j < W; j++) {
            C[i + 1][j + 1] = c[j];
        }
    }
    if (H == 1 && W == 1) {
        prn("No");
        return;
    }
    for (int i = 1; i <= H; i++) {
        for (int j = 1; j <= W; j++) {
            if (C[i][j] == '#' && C[i - 1][j] != '#' && C[i + 1][j] != '#' && C[i][j - 1] != '#' && C[i][j + 1] != '#') {
                prn("No");
                return;
            }
        }
    }
    prn("Yes");
}

void prn(String s) {
    out.println(s);
}

int ni() {
    return in.nextInt();
}

String ns() {
    return in.nextString();
}

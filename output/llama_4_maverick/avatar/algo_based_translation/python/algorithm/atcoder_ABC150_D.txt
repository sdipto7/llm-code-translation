1. INPUT n, num
2. INPUT num_set as a set of integers
3. INITIALIZE two_times_set as an empty set
4. FOR EACH i IN num_set
   4.1 INITIALIZE j as 1
   4.2 WHILE j <= 30
      4.2.1 i = i DIV 2
      4.2.2 j = j + 1
      4.2.3 IF i MOD 2 != 0 THEN
         4.2.3.1 ADD j-1 TO two_times_set
         4.2.3.2 BREAK
   4.3 IF SIZE(two_times_set) != 1 THEN
      4.3.1 OUTPUT 0
      4.3.2 EXIT
5. CONVERT num_set TO num_list
6. INITIALIZE lcm as num_list[0]
7. FOR i FROM 1 TO SIZE(num_list)-1
   7.1 lcm = lcm * num_list[i] DIV GCD(lcm, num_list[i])
8. OUTPUT (num - lcm DIV 2) DIV lcm + 1

3. INITIALIZE h1 = 0
4. INITIALIZE h2 = 0
5. INITIALIZE x = 0
6. FOR i FROM 0 TO n-1
7.     IF number[i] NOT EQUAL TO 4 AND number[i] NOT EQUAL TO 7
8.         PRINT "NO"
9.         INCREMENT x BY 1

14.        ADD number[i] TO h2
15. IF x EQUAL TO 0

1. READ input string soz
2. INITIALIZE array a of size length(soz) + 1 with all elements as 0
3. FOR i FROM 1 TO length(soz)
   3.1. SET a[i] = a[i - 1]
   3.2. IF soz[i - 1] == soz[i] THEN
      3.2.1. INCREMENT a[i] by 1
4. READ integer n
5. INITIALIZE empty array arr
6. FOR i FROM 1 TO n
   6.1. READ integers m and l
   6.2. APPEND a[l - 1] - a[m - 1] to arr
7. FOR EACH element i in arr
   7.1. PRINT i

```python
# Function signature
def main():
    # Variable declarations
    import sys
    n: int = int(sys.stdin.readline())
    c: int = 0
    flg: int = 0

    # Logic flow
    while n > 0:
        x, y = map(int, sys.stdin.readline().split())
        if x == y:
            c += 1
            if c >= 3:
                flg = 1
        else:
            c = 0
        n -= 1

    if flg == 0:
        print("No")
    else:
        print("Yes")

# Dependencies
# Requires Python sys module for input handling
```

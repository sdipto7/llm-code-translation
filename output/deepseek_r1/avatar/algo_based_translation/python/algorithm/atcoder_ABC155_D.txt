#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int N, K;
    cin >> N >> K;
    vector<long long> A(N);
    for (int i = 0; i < N; ++i) {
        cin >> A[i];
    }
    sort(A.begin(), A.end());
    
    long long ll = -1000000000000000001LL;
    long long rr = 1000000000000000001LL;
    
    while (ll + 1 < rr) {
        long long x = (ll + rr) / 2;
        long long tot = 0;
        for (int i = 0; i < N; ++i) {
            if (A[i] < 0) {
                long long l = -1, r = N;
                while (l + 1 < r) {
                    long long c = (l + r) / 2;
                    __int128_t product = (__int128_t)A[i] * A[c];
                    if (product < x) {
                        r = c;
                    } else {
                        l = c;
                    }
                }
                tot += N - r;
            } else {
                long long l = -1, r = N;
                while (l + 1 < r) {
                    long long c = (l + r) / 2;
                    __int128_t product = (__int128_t)A[i] * A[c];
                    if (product < x) {
                        l = c;
                    } else {
                        r = c;
                    }
                }
                tot += r;
            }
            __int128_t square = (__int128_t)A[i] * A[i];
            if (square < x) {
                tot--;
            }
        }
        tot /= 2;
        if (tot < K) {
            ll = x;
        } else {
            rr = x;
        }
    }
    cout << ll << endl;
    return 0;
}

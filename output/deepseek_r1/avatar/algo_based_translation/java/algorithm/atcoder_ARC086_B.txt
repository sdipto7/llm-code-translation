```python
# Function signature
def main():
    # Variable declarations
    n: int
    a: list[int]
    r: list[str] = []
    m: int

    # Input operations
    n = int(input())
    a = list(map(int, input().split()))

    # Logic flow
    m = 0
    for i in range(1, n):
        if abs(a[m]) < abs(a[i]):
            m = i

    for i in range(n):
        if (a[m] >= 0) ^ (a[i] >= 0):
            r.append(f"{m+1} {i+1}")

    if a[m] >= 0:
        for i in range(1, n):
            r.append(f"{i} {i+1}")
    else:
        for i in range(n, 1, -1):
            r.append(f"{i} {i-1}")

    # Output operations
    print(len(r))
    print('\n'.join(r))

# Dependencies
# Requires Python built-ins, no external modules
```

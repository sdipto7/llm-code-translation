```python
def main():
    import sys
    data = sys.stdin.read().split()
    n = int(data[0])
    maxn = 4010
    a = int(data[1])
    b = int(data[2])
    c = int(data[3])
    arr = [a, b, c]
    dp = [0] * maxn
    for i in range(n):
        dp[i] = -10**9
    dp[0] = 0
    for i in range(3):
        for j in range(arr[i], maxn):
            dp[j] = max(dp[j], dp[j - arr[i]] + 1)
    print(dp[n])
```

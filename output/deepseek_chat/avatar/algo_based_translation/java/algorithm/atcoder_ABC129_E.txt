Scanner input = System.in
int MOD = 1000000007
char[] s = input.nextLine().toCharArray()
int n = s.length
int[][] dp = new int[n+1][2]
dp[0][0] = 1

for i from 0 to n-1:
    if s[i] == '0':
        dp[i+1][0] = dp[i][0]
        dp[i+1][1] = dp[i][1]
    else:
        dp[i+1][1] = (dp[i][0] + dp[i][1]) % MOD
    
    if s[i] == '0':
        dp[i+1][1] = (dp[i+1][1] + dp[i][1] * 2) % MOD
    else:
        dp[i+1][0] = (dp[i+1][0] + dp[i][0] * 2) % MOD
        dp[i+1][1] = (dp[i+1][1] + dp[i][1] * 2) % MOD

int result = (dp[n][0] + dp[n][1]) % MOD
print result

mod = 1000000007
n = integer input
k = integer input

len = empty list
l = 1
while l <= n:
    r = n // (n // l)
    append (r - l + 1) to len
    l = r + 1

q = length of len
dp = array of zeros with length = k * (q + 1)

for j from 1 to q:
    dp[j] = len[j-1] + dp[j-1]

for i from 1 to k-1:
    for j from 1 to q:
        index1 = i * (q + 1) + j
        index2 = i * (q + 1) + j - 1
        index3 = (i - 1) * (q + 1) + q - j + 1
        dp[index1] = dp[index2] + dp[index3] * len[j-1]
        dp[index1] = dp[index1] % mod

output dp[k * (q + 1) - 1]

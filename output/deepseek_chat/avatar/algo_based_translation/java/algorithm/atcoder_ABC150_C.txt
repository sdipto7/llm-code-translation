Algorithm:
- Import java.util.ArrayList, java.util.Arrays, java.util.List, java.util.Scanner

Function permutation(List<String> list, String target, String ans):
  If target.length() <= 1:
    list.add(ans + target)
  Else:
    For i from 0 to target.length() - 1:
      Call permutation(list, target.substring(0, i) + target.substring(i + 1), ans + target.charAt(i))
  Return list

Function main(String[] args):
  Initialize Scanner scanner with System.in
  Read integer n from scanner
  Create 2x1 String array line
  For i from 0 to 1:
    Set line[i][0] to empty string
    For j from 0 to n-1:
      Append next token from scanner to line[i][0]
  
  Initialize empty string number
  For i from 1 to n:
    Append string representation of i to number
  
  Initialize empty ArrayList<String> listA
  Call permutation(listA, number, "")
  
  Initialize integer sum = 0
  For j from 0 to line.length - 1:
    For i from 0 to listA.size() - 1:
      If listA.get(i) equals line[j][0]:
        If sum == 0:
          Add i to sum
        Else:
          Subtract i from sum
  
  Print absolute value of sum

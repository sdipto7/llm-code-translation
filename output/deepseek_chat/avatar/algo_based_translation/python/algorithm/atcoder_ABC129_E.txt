mod = 1000000007
l = input string
n = length of l
dp = 2D array of size (n+1) x 2, initialized to zeros
dp[0][0] = 1
dp[0][1] = 0

for i from 0 to n-1:
    c = l[i]
    if c == '0':
        dp[i+1][0] = dp[i+1][0] + dp[i][0]
        dp[i+1][1] = dp[i+1][1] + dp[i][1] * 3
    else:
        dp[i+1][0] = dp[i+1][0] + dp[i][0] * 2
        dp[i+1][1] = dp[i+1][1] + dp[i][0]
        dp[i+1][1] = dp[i+1][1] + dp[i][1] * 3
    
    dp[i+1][0] = dp[i+1][0] % mod
    dp[i+1][1] = dp[i+1][1] % mod

result = (dp[n][0] + dp[n][1]) % mod
print result

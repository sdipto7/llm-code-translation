read():
    Read integer 'number' from input
    If 'number' equals 0:
        Return false
    Else:
        Create integer array 'score' of size 'number'
        For i from 0 to number-1:
            Read integer into score[i]
        Return true

main():
    While read() returns true:
        Call slove()

slove():
    Sort 'score' array in ascending order
    Initialize integer 'sum' to 0
    For i from 1 to number-2:
        Add score[i] to sum
    Calculate average = sum / (number-2)
    Print average

Required: java.util.Scanner, java.util.Arrays

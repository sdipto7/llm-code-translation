```python
import sys

def main():
    sc = sys.stdin
    DEVISOR = 1000000007
    n = int(sc.readline().strip())
    k = int(sc.readline().strip())
    ans = 0
    i = k
    while i <= n + 1:
        max_val = i * (2 * n - i + 1) // 2
        min_val = i * (i - 1) // 2
        ans = (ans + (max_val - min_val + 1) % DEVISOR) % DEVISOR
        i += 1
    print(ans % DEVISOR)
```

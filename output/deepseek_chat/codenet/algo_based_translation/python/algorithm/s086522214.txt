import numpy as np

def prime_decomposition(n: int) -> list[int]:
    i = int(2)
    table = list()
    while i * i <= n:
        while n % i == 0:
            n = n / i
            table.append(int(i))
        i = i + 1
    if n > 1:
        table.append(int(n))
    return table

def digit(i: int) -> list[int]:
    if i > 0:
        return digit(i // 10) + [i % 10]
    else:
        return list()

def getNearestValueIndex(list: list[float], num: float) -> int:
    idx = np.abs(np.asarray(list) - num).argmin()
    return idx

def calculate_iterations(a: int, b: int) -> int:
    hole: int = 1
    i: int = 0
    check: bool = True
    while check:
        if hole >= b:
            check = False
        else:
            hole = a * (i + 1) - i
            i += 1
    return i

# Required Libraries: None
# Input: Two integers, a and b, read from standard input
# Output: Print the integer result from calculate_iterations function

def main():
    a, b = map(int, input().split())
    result = calculate_iterations(a, b)
    print(result)

def determine_winner() -> None:
    # Read input and convert to list of characters
    s: List[str] = list(input())

    # Initialize variables
    x: bool = True
    dt: Dict[str, int] = {}

    # Count occurrences of each character
    for i in s:
        if i in dt:
            dt[i] += 1
        else:
            dt[i] = 1

    # Check number of odd occurrences
    cnt: int = 0
    x = True
    for key, value in dt.items():
        if value % 2 != 0 and cnt < 1:
            cnt += 1
        elif value % 2 != 0 and cnt >= 1:
            x = False
            break

    # Determine and print the winner
    if x:
        print("First")
    elif not x and len(s) % 2 == 0:
        print("Second")
    elif not x and len(s) % 2 != 0:
        print("First")

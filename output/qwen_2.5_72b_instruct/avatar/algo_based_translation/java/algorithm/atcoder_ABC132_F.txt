Algorithm:
function main() -> void
    variable mod: long = 1000000007
    variable sc: Scanner = new Scanner(System.in)
    variable n: int = sc.nextInt()
    variable k: int = sc.nextInt()
    sc.close()
    variable len: ArrayList<Integer> = new ArrayList<Integer>()
    variable l: int = 1
    while l <= n
        variable r: int = n / (n / l)
        len.add(r - l + 1)
        l = r + 1
    variable q: int = len.size()
    variable dp: long[] = new long[k * (q + 1)]
    for j from 1 to q
        dp[j] = len.get(j - 1) + dp[j - 1]
    for i from 1 to k - 1
        for j from 1 to q
            dp[i * (q + 1) + j] = dp[i * (q + 1) + j - 1] + dp[(i - 1) * (q + 1) + q - j + 1] * len.get(j - 1)
            dp[i * (q + 1) + j] %= mod
    print(dp[k * (q + 1) - 1])

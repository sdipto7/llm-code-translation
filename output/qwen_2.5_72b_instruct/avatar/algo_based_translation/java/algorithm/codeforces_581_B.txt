TreeSet<Long> set

function main(String[] args)
    Scanner in
    PrintWriter pw
    int n = in.nextInt()
    long[] a = new long[n]
    for int i from 0 to n-1
        a[i] = in.nextLong()
    long[] dp = new long[n+1]
    Arrays.fill(dp, -1)
    dp[n-1] = a[n-1]
    for int i from n-1 to 0
        dp[i] = Math.max(dp[i+1], a[i])
    for int i from 0 to n-1
        if a[i] > dp[i+1]
            pw.print(0 + " ")
        else
            pw.print((dp[i+1] - a[i] + 1) + " ")
    pw.println()
    pw.close()

function debug(Object... obj)
    System.err.println(Arrays.deepToString(obj))

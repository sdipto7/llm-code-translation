main(String[] args)
    input: Scanner s
    String s1 = s.next()
    String s2 = s.next()
    if isValid(s1, s2) is false
        output: -1
    else
        HashMap<Character, Integer> m1 = new HashMap<>()
        HashMap<Character, Integer> m2 = new HashMap<>()
        for char c in s1.toCharArray()
            if m1.containsKey(c)
                m1.put(c, m1.get(c) + 1)
            else
                m1.put(c, 1)
        for char c in s2.toCharArray()
            if m2.containsKey(c)
                m2.put(c, m2.get(c) + 1)
            else
                m2.put(c, 1)
        int ans = 0
        HashSet<Character> hs = new HashSet<>()
        for char c in s2.toCharArray()
            if !hs.contains(c)
                hs.add(c)
                int x1 = m1.get(c)
                int x2 = m2.get(c)
                x1 = Math.min(x1, x2)
                ans += x1
        output: ans

isValid(String s1, String s2)
    int n1 = s1.length()
    int n2 = s2.length()
    HashMap<Character, Boolean> map = new HashMap<>()
    for char c in s1.toCharArray()
        map.put(c, true)
    for char c in s2.toCharArray()
        if !map.containsKey(c)
            return false
    return true

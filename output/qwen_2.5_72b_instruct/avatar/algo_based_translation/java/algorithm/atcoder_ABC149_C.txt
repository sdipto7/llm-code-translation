function main() {
    var stdIn: Scanner
    var x: int

    stdIn = new Scanner(System.in)
    x = Integer.parseInt(stdIn.next())

    if (x != 2 && x % 2 == 0) {
        x++
    }

    while (true) {
        var i: int
        var sqrtNum: double

        sqrtNum = Math.sqrt(x)

        for (i = 3; i <= sqrtNum; i += 2) {
            if (x % i == 0) {
                break
            }
        }

        if (i > sqrtNum) {
            break
        }

        x += 2
    }

    System.out.println(x)
}

Function main(String[] args) throws IOException
    BufferedReader br = new BufferedReader(new InputStreamReader(System.in))
    PrintWriter pw = new PrintWriter(new OutputStreamWriter(System.out))
    Fast f = new Fast()
    f.sol(br, pw)
    pw.close()

Function sol(BufferedReader br, PrintWriter pw) throws IOException
    StringTokenizer st = new StringTokenizer(br.readLine())
    String s = st.nextToken()
    Integer[] cum = new Integer[s.length() + 1]
    cum[0] = 0
    cum[s.length()] = 0
    for int i = 1; i < s.length(); i++
        if s.charAt(i) == s.charAt(i - 1)
            cum[i] = cum[i - 1] + 1
        else
            cum[i] = cum[i - 1]
    st = new StringTokenizer(br.readLine())
    Integer q = Integer.parseInt(st.nextToken())
    while q-- != 0
        st = new StringTokenizer(br.readLine())
        Integer l = Integer.parseInt(st.nextToken())
        Integer r = Integer.parseInt(st.nextToken()) - 1
        pw.println(cum[r] - cum[l - 1])

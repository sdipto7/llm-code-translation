function main():
    n: int = input()
    t: List[int] = map(int, input().split())
    m: int = input()
    for i in range(m):
        p: int, x: int = map(int, input().split())
        s: int = 0
        for j in range(len(t)):
            if j == p - 1:
                s += x
            else:
                s += t[j]
        print(s)
